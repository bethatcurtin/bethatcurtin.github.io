4. TikZ Diagram of a Neural Network (LaTeX Code)

\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{positioning}

\begin{document}

\begin{tikzpicture}[node distance=1cm and 1.5cm, every node/.style={circle, draw, minimum size=1cm}]
  % Input layer
  \node (i1) {I1};
  \node[below=of i1] (i2) {I2};
  \node[below=of i2] (i3) {I3};

  % Hidden layer
  \node[right=of i1] (h1) {H1};
  \node[below=of h1] (h2) {H2};
  \node[below=of h2] (h3) {H3};

  % Output layer
  \node[right=of h2] (o1) {O1};

  % Connections
  \foreach \i in {i1,i2,i3}
    \foreach \h in {h1,h2,h3}
      \draw[->] (\i) -- (\h);

  \foreach \h in {h1,h2,h3}
    \draw[->] (\h) -- (o1);

\end{tikzpicture}

\end{document}
